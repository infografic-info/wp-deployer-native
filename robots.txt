# Block bots from accessing author, category, and tag pages to avoid duplicate content
User-agent: *
Disallow: /author
Disallow: /category
Disallow: /tag

# Block URLs related to Elementor Library, XML-RPC, and WordPress JSON API
Disallow: /?elementor_library=
Disallow: /xmlrpc.php
Disallow: /wp-json/

# Block feeds and comments to avoid duplicate content and unnecessary pages
Disallow: /*/feed/
Disallow: /*/comments/

# Block URLs with search parameters and internal search pages
Disallow: /*?*
Disallow: /?s=
Disallow: /search/

# Block access to WordPress admin folder except for admin-ajax.php, needed for some functionalities
User-agent: *
Disallow: /wp-admin/
Allow: /wp-admin/admin-ajax.php

# Inform search engines about the sitemap location
Sitemap: https://domain.com/sitemap_index.xml
